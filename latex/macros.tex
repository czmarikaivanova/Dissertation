\newcommand{\includepaperpages}[2]
{
	\forloop{ct}{0}{\value{ct} < #2}
	{
		\begin{figure}[ht]
			\includegraphics[width=.95\textwidth]{#1\the\value{ct}.ps}
		\end{figure}
		\clearpage
	}
}
